Exercise 1: Create Odd-Even Dictionary

This exercise creates a dictionary where numbers from 0 to 99 are labeled as either "odd" or "even".

def create_odd_even_dict():
 result_dict = {}
 for i in range(0, 100):
  if i % 2 == 0:
   result_dict[i] = 'even'
  else:
   result_dict[i] = 'odd'
 return result_dict
result = create_odd_even_dict()
print(result)

Exercise 4: Prime Number Checker

This exercise defines a function to check if a number is prime and also prints all prime numbers less than 100 using list comprehension.

def isprime(n):
 c = 0
 if n > 2:
  for j in range(2, 101):
   if n == j:
    continue
   elif j < n:
    if n % j == 0:
     c = 0
     break
    else:
     c = 1
 if c == 1:
  return True
 elif c == 0:
  return False
 elif n == 1 or n == 2:
  return True

num = int(input())
res = isprime(num)
if res == True:
 print(num, "is a prime number!")
elif res == False:
 print(num, "is not a prime number!")

primelist = [i for i in range(2, 101) if all(i % j != 0 for j in range(2, i))]
print("Prime numbers less than 100:", primelist)
